function checkAnswer() { const current = writingVocabulary[currentIndex]; const userVal = inputElem.value.trim().toLowerCase(); const correctVal = current.word.trim().toLowerCase(); const feedback = document.getElementById('feedback-msg'); const actionBtn = document.getElementById('action-btn'); if (!userVal) return; if (userVal === correctVal) { // Poprawna odpowiedź isAnswered = true; inputElem.disabled = true; inputElem.className = 'input-field correct'; feedback.textContent = '✨ Świetnie! Poprawna odpowiedź.'; feedback.className = 'feedback-msg correct'; showExampleBox(current.examplePL, current.exampleEN); actionBtn.textContent = 'Następne słowo →'; actionBtn.setAttribute('onclick', 'nextWord()'); } else { // Błędna odpowiedź attempts++; inputElem.className = 'input-field incorrect'; if (attempts === 1) { // Pierwszy błąd -> podajemy pierwszą literę w nowym formacie const firstLetter = current.word.charAt(0).toUpperCase(); feedback.textContent = `💡 Zaczyna się na "${firstLetter}"`; feedback.className = 'feedback-msg incorrect'; inputElem.focus(); inputElem.select(); } else if (attempts >= 2) { // Drugi błąd -> podajemy pełną odpowiedź i odblokowujemy przejście dalej isAnswered = true; inputElem.disabled = true; feedback.textContent = `❌ Poprawna odpowiedź to: "${current.word}"`; feedback.className = 'feedback-msg incorrect'; showExampleBox(current.examplePL, current.exampleEN); actionBtn.textContent = 'Następne słowo →'; actionBtn.setAttribute('onclick', 'nextWord()'); } } }
